\noindent\textbf{Problem 1. (5.0 points)} 
\textbf{a)} Given the function $u(x, y) = (x + y)\sqrt{x^2 + y^2}$. Prove that:
\[ x\frac{\partial u}{\partial x} + y\frac{\partial u}{\partial y} = 2u \quad \text{for all } (x, y) \neq (0,0). \]

\noindent\textbf{b)} Find the extrema of the function: $z = (x + y)^2 - \frac{x^4}{2} - \frac{y^4}{2}$.

\vspace{0.2cm}
\noindent\textbf{c)} Find the extrema of the function: $z = 20x^{\frac{1}{4}}y^{\frac{3}{4}}$, subject to the constraint $x + 3y = 8 \ (x, y > 0)$.

\vspace{0.3cm}
\noindent\textbf{Problem 2. (3.0 points)} 
\textbf{a)} Calculate the area of the plane region $D$ bounded by the curves
\[ y = \frac{1}{x}; \ y = \frac{5}{x}; \ x = y^2 \text{ and } x = 5y^2. \]

\noindent\textbf{b)} The population density of a city is given by the function $f(x, y) = 20000e^{\frac{-(x^2+y^2)}{5}}$, calculated on the coordinate plane $Oxy$ with the origin as the city center (unit: people/km$^2$). Calculate the population of the city in the region with a radius of 5 km from the city center. What is the average population density of this region?

\vspace{0.3cm}
\noindent\textbf{Problem 3. (1.0 point)} In the space $\mathbb{R}^3$, consider the object:
\[ \Omega = \left\{ (x, y, z) \in \mathbb{R}^3 : \sqrt{x^2 + y^2} \le z \le 4 - \sqrt{x^2 + y^2} \right\} \]
with mass density distribution given by the function $\rho(x, y, z) = x^2 + y^2$. Calculate the mass of $\Omega$.

\vspace{0.3cm}
\noindent\textbf{Problem 4. (1.0 point)} Calculate the line integral
\[ I = \int_\gamma (5y - 6^x)dx + (10x + 6^y)dy, \]
where $\gamma$ is the circle $x^2 + y^2 = 36$.
